// parse_demo.hl — parse an .hl file and walk the locus tower.

import "vendor/pond/heron" as heron;

fn main() {
    if std::env::args_count() < 2 {
        println("usage: parse_demo <source.hl>");
        return;
    }

    let path = std::env::arg(1);
    let content = std::io::fs::read_file(path) or raise;

    let parser = heron::Parser { };
    if !parser.is_ready() {
        println("parser allocation failed");
        return;
    }

    let tree = heron::parse(parser, content);
    println("root kind: ", heron::node_kind(tree.root()));
}
